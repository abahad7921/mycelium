00051 void Url::assign(const string& s)
00052 {
00053     this->clear();
00054     enum {
00055         URL = 0,
00056         SCHEME_CLN = 1,
00057         SCHEME    = 2,
00058         DSLASH_AUTH = 3,
00059         AUTHORITY = 4,
00060         PATH    = 5,
00061         QUERY    = 6,
00062         FRAGMENT = 7
00063     };
00064 
00065     try {
00066         //regex re(url_regex[RE_URL],regex_constants::perl);
00067         smatch match;
00068         if( regex_match(s, match, url_re) ) {
00069             /* bad cases:
00070              * ////rrara
00071              * 3: //
00072              * 5: //rara
00073              * then is this url an absolute path like /rara  ?
00074              * so rule of thumb is to sanityze double slashes on paths.
00075              */
00076             if( match[SCHEME].matched )
00077                 scheme(match[SCHEME].str());
00078 
00079             if( match[DSLASH_AUTH].matched ) {
00080                 // Only valid for locally scoped urls like file:/// that define a default host, since http:/// makes no sense, authority must exists for globally scoped urls
00081                 // in this case 3: // 4 matches, but its empty
00082                 // if url == "/////" then 3: // 4 matches but empty 5: /// we treat this like an absolute path, but only if there's no scheme.
00083                 if( match[DSLASH_AUTH].str() == "//") {
00084                     // if( scheme().empty() )
00085                     // this is a relative uri with an absolute path we are ok.
00086                     if ( ! scheme().empty() && scheme() != "file" )
00087                         throw UrlParseError("empty authority part, with // is not allowed for schemes other than 'file'");
00088                 }
00089             }
00090 
00091             if( match[AUTHORITY].matched )
00092                 authority(match[AUTHORITY].str());
00093             if( match[PATH].matched )
00094                 //DLOG(cout << "PATH match: |" << match[PATH].str() << "|" << endl);
00095                 path(match[PATH].str());
00096 
00097             if( match[QUERY].matched )
00098                 query(match[QUERY].str());
00099 
00100             if( match[FRAGMENT].matched )
00101                 fragment(match[FRAGMENT].str());
00102 
00103             if( ! syntax_ok() )
00104                 throw UrlParseError(" ! syntax_ok() for this url, sanity checks failed");
00105 
00106             if( ! valid_host())
00107                 throw UrlParseError(" ! valid_host() for this url, sanity checks failed");
00108 
00109         } else {
00110             throw UrlParseError("Url doesn't match primary url regex");
00111         }
00112     } catch(regex_error re) {
00113         string s("Url::assign() boost::regex_error: ");
00114         s+=re.what();

00421                 assert(host_e - host_b > 0 );
