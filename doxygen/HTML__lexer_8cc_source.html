00364         } else if ( tok == "noindex" )

00366 
00367         else if ( tok == "nofollow" )   
00368             analysis->follow = false;
00369 
00370         else if ( tok == "none" ) {  
00371             analysis->follow = false;
00372             analysis->index = false;
